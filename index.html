void qxx(int x,int y){
cnt++;nxt[cnt]=head[x];head[x]=cnt;to[cnt]=y;
}
int get_lenth(int x,int fa){
int sum1=0,sum2=0;
for(register int i=head[x];i;i=nxt[i]){
int k=to[i];
if(k==fa) continue;
sum2=max(sum2,get_lenth(k,x)+1);
if(sum2>sum1){
swap(sum1,sum2);
}
}
maxv=max(maxv,sum1+sum2);
return sum1;
}
